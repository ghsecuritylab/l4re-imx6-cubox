260 L4_INLINE l4_msgtag_t
261 l4_irq_attach_u(l4_cap_idx_t irq, l4_umword_t label,
262  l4_cap_idx_t thread, l4_utcb_t *utcb) L4_NOTHROW
263 {
264  int items = 0;
265  l4_msg_regs_t *m = l4_utcb_mr_u(utcb);
266  m->mr[0] = L4_IRQ_SENDER_OP_ATTACH;
267  m->mr[1] = label;
268 
269  if (!l4_is_invalid_cap(thread))
270  {
271  items = 1;
272  m->mr[2] = l4_map_obj_control(0, 0);
273  m->mr[3] = l4_obj_fpage(thread, 0, L4_FPAGE_RWX).raw;
274  }
275  return l4_ipc_call(irq, utcb, l4_msgtag(L4_PROTO_IRQ_SENDER, 2, items, 0),
276  L4_IPC_NEVER);
277 }
