596  for (; *p && !__is_sep(*p) && *n; ++p, ++n)

630 inline Path
631 Mount_tree::lookup(Path const &path, cxx::Ref_ptr<Mount_tree> *mt,
632  cxx::Ref_ptr<Mount_tree> *mp) throw()
633 {
634  cxx::Ref_ptr<Mount_tree> x(this);
635  Path p = path;
636 
637  if (p.first().cmp_path("."))
638  p.strip_first();
639 
640  Path last_mp = p;
641 
642  if (mp)
643  *mp = x;;
644 
645  while (1)
646  {
647  Path r = x->find(p, &x);
648 
649  if (!r)
650  {
651  if (mp)
652  return last_mp;
653 
654  if (mt)
655  *mt = x;
656 
657  return p;
658  }
659 
660  r.strip_sep();
661 
662  if (mp && x->_mount)
663  {
664  last_mp = r;
665  *mp = x;
666  }
667 
668  if (r.empty())
669  {
670  if (mt)
671  *mt = x;
672 
673  if (mp)
674  return last_mp;
675  else
676  return r;
677  }
678 
679  p = r;
680  }
